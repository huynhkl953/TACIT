run_seurat_clustering <- function(data_matrix) {
  data_matrix_t <- t(data_matrix)
  rownames(data_matrix_t) <- colnames(data_matrix)
  colnames(data_matrix_t) <- rownames(data_matrix)
  
  seurat_obj <- CreateSeuratObject(counts = data_matrix_t)
  seurat_obj <- NormalizeData(seurat_obj, normalization.method = "CLR", margin = 2)
  seurat_obj <- FindVariableFeatures(seurat_obj, nfeatures = ncol(data_matrix))
  seurat_obj <- ScaleData(seurat_obj, features = rownames(seurat_obj))
  seurat_obj <- RunPCA(seurat_obj, features = VariableFeatures(object = seurat_obj))
  seurat_obj <- FindNeighbors(seurat_obj, dims = 1:2)
  seurat_obj <- FindClusters(seurat_obj, resolution = 20) # High resolution to get many clusters
  
  return(seurat_obj)
}

threshold_groups <- function(input_vector, thresholds) {
  # Initialize an empty vector to hold the group assignments
  groups <- numeric(length(input_vector))
  
  # Loop through each threshold in the list
  for (i in seq_along(thresholds)) {
    # Create a logical vector indicating which elements in the input_vector are above the threshold
    group <- input_vector >= thresholds[i]
    
    # Assign group i to all elements in input_vector that are above threshold i and below threshold (i+1)
    if (i < length(thresholds)) {
      next_group <- input_vector >= thresholds[i + 1]
      group <- group & !next_group
    }
    
    # Assign group i to all elements in input_vector that are above threshold i
    groups[group] <- i
  }
  
  # Return the vector of group assignments
  return(groups)
}

assign_expression_groups <- function(input_vector, thresholds) {
  # Sort thresholds to ensure correct group assignment
  thresholds <- sort(unique(thresholds))
  # Use the cut function to create groups. `findInterval` is also a good choice.
  # The labels are 0-indexed to represent Group 0, Group 1, etc.
  group_labels <- findInterval(input_vector, thresholds)
  return(group_labels)
}

threshold_function_negative <- function(k, data_anb, clusters) {
  ## Save the name of the antibody column
  name <- colnames(data_anb[k])

  ## Combine intensity data with Louvain cluster assignments
  data_intensity_cluster <- data.frame(data_anb, lv_clusters = clusters)

  # Order Louvain clusters by the median intensity of the current antibody
  group_ordered <- with(
    data_intensity_cluster,
    reorder(
      lv_clusters,
      get(name), median
    )
  )

  # Create a data frame with median values for each cluster, ordered from lowest to highest
  data_median <- data.frame(median_value = attr(group_ordered, "scores"), lv_clusters = names(attr(group_ordered, "scores")))
  data_median <- data_median[order(data_median$median_value, decreasing = F), ]

  ## Handle the edge case where all median values are zero
  if (sum(data_median$median_value) == 0) {
    threshold_point_final <- max(data_intensity_cluster[, name])
  } else {
    # --- Piecewise Regression to find breakpoints ---
    x <- 1:nrow(data_median)
    y <- data_median$median_value
    fit <- lm(y ~ x)

    # Find the optimal number of breakpoints (1, 2, or 3) by checking AIC
    AIC_value <- c()
    for (i in 1:3) {
      out <- tryCatch(
        { segmented(fit, seg.Z = ~x, npsi = i) },
        error = function(e) { e }
      )
      if (inherits(out, "error")) {
        AIC_value[i] <- 100000
      } else {
        AIC_value[i] <- AIC(out)
      }
    }

    ## Fit the final model with the number of breakpoints that gave the lowest AIC
    num_breakpoints <- which.min(AIC_value)
    segmented.fit <- segmented(fit, seg.Z = ~x, npsi = num_breakpoints)

    # --- DEFINE GROUPS AND RUN GRID SEARCH OPTIMIZATION ---

    # Get the breakpoint x-values (cluster indices) and sort them
    breakpoints <- sort(segmented.fit$psi[, 2])

    # Assign all ordered clusters to an expression group (0, 1, 2, etc.)
    # based on the breakpoints found.
    Group_expression <- assign_expression_groups(1:nrow(data_median), thresholds = breakpoints)
    data_median_update <- data.frame(data_median, Group_expression)
    new_data_anb <- merge(data_intensity_cluster, data_median_update, "lv_clusters")

    # Define the two populations for comparison to find the negative threshold:
    # "low_data" = cells from the lowest intensity clusters (Group 0)
    # "high_data_for_neg_cutoff" = cells from the second-lowest intensity clusters (Group 1)
    low_data <- new_data_anb[which(new_data_anb$Group_expression == 0), name]
    high_data_for_neg_cutoff <- new_data_anb[which(new_data_anb$Group_expression == 1), name]
    
    # Check if both groups have cells to compare
    if(length(low_data) == 0 || length(high_data_for_neg_cutoff) == 0){
        # If one group is empty, a robust threshold cannot be calculated.
        # Fallback to using the intensity value at the first breakpoint.
        breakpoint_ys <- segmented.fit$fitted.values[floor(breakpoints)]
        threshold_point_final <- breakpoint_ys[1]
    } else {
        # Perform the grid search to find the optimal threshold between Group 0 and Group 1
        # The search space is defined by the quantiles of the combined data from the two groups
        search_range <- c(low_data, high_data_for_neg_cutoff)
        cut_seq <- seq(as.numeric(quantile(search_range, 0.05)), as.numeric(quantile(search_range, 0.95)), length.out = 1000)
        cut_seq <- cut_seq[cut_seq > 0] # Ensure we don't use 0

        low_high <- high_low <- c()
        for (g in 1:length(cut_seq)) {
          # low_high: proportion of "high" (Group 1) cells below the current cutoff
          low_high[g] <- sum(high_data_for_neg_cutoff <= cut_seq[g]) / length(high_data_for_neg_cutoff)
          # high_low: proportion of "low" (Group 0) cells above the current cutoff
          high_low[g] <- sum(low_data > cut_seq[g]) / length(low_data)
        }
        # The final threshold is the cutoff that minimizes the sum of these two error rates
        threshold_point_final <- cut_seq[which.min(low_high + high_low)]
    }
  }

  ## Binarize the original data based on the final threshold
  Group_Threshold_new <- ifelse(as.numeric(t(data_intensity_cluster[, name])) > threshold_point_final, 1, 0)

  ## Return the binarized vector and the calculated threshold value
  return(list(Group_Threshold_new, threshold_point_final))
}

Scenario 2: Mixture Normal with Unequal Variance In this scenario, the high expression group is more spread out than the low expression group, which is common in biological data. We will use a “low” group from N(mu=2,sigma=0.5) and a “high” group from N(mu=4,sigma=1.25). This makes it more challenging to find a clean cutoff.

# Calculate thresholds and generate plots for each marker
plots_uneq <- list()
threshold_results_uneq <- data.frame()

for (k in 1:n_features) {
  marker_name <- colnames(sim_data_unequal_var)[k]
  
  # Calculate thresholds
  negative_res <- threshold_function_negative(k, sim_data_unequal_var, scfp_uneq$seurat_clusters)
  
  neg_thresh <- negative_res[[2]]
  
  threshold_results_uneq <- rbind(threshold_results_uneq, data.frame(Marker = marker_name, Negative_Threshold = neg_thresh))
  
  # Prepare data for plotting
  data_plot <- data.frame(marker = sim_data_unequal_var[, k], Cluster = scfp_uneq$seurat_clusters)
  ranked_clusters <- data_plot %>%
    group_by(Cluster) %>%
    summarise(mean_marker = mean(marker), .groups = 'drop') %>%
    arrange(mean_marker)
  
  # Create the plot
  plots_uneq[[k]] <- ggplot(ranked_clusters, aes(x = reorder(Cluster, mean_marker), y = mean_marker)) +
    geom_point(color = "purple4", size = 2) +
    geom_hline(yintercept = neg_thresh, linetype = "dashed", color = "blue") +
    annotate("text", x = Inf, y = neg_thresh, label = paste("Neg:", round(neg_thresh, 2)), hjust = 1.1, vjust = 1.5, color="blue") +
    labs(
      title = paste("Thresholds for", marker_name),
      subtitle = "Unequal Variance (μ=2, σ=0.5 vs μ=4, σ=1.25)",
      x = "Microcluster (Ranked by Mean Expression)",
      y = "Mean Expression"
    ) +
    theme_classic(base_size = 10) +
    theme(axis.text.x = element_text(angle = 90, vjust = 0.5, size=6))
}
